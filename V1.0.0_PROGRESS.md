```
src/main/
├── index.ts          # 主进程入口，窗口管理，自动扫描
└── preload.ts        # IPC 桥接，安全 API 暴露
```

```
src/renderer/
├── index.html        # HTML 入口
├── main.tsx          # React 入口
├── App.tsx           # 主应用
├── index.css         # 全局样式
```
